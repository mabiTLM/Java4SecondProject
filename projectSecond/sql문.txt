use nano_test;


drop table future_prospective;
drop table alumni;
drop table honors_and_awards;
drop table careers;
drop table university;
drop table members;
drop table news_board;

create table news_board (
id int auto_increment primary key,
category varchar(20) not null,
title varchar(50) not null,
content longtext not null,
created_at Date default (current_date),
views int default 0
);

create table members(
id int auto_increment primary key,
email varchar(200) not null unique,
user_id varchar(200) not null unique,
password varchar(100) not null,
category varchar(50),
english_name varchar(30),
name varchar(20),
position varchar(20),
info varchar(300),
image longtext,
is_manager tinyint,
chinese_name varchar(20),
address varchar(200),
phone varchar(15),
site varchar(100),
link longtext
);

create table university(
id int auto_increment primary key,
content varchar(300) not null unique
);

create table careers(
id int auto_increment primary key,
university int,
member_id int,
title varchar(200),
content varchar(200),
start date,
end date,
foreign key (university) references university(id),
foreign key (member_id) references members(id)
);

create table honors_and_awards(
id int auto_increment primary key,
member_id int,
title varchar(300) not null,
award_date date,
contents varchar(300)
);

create table alumni(
id int auto_increment primary key,
member_id int,
content varchar(200),
start date,
end date
);

create table future_prospective(
id int auto_increment primary key,
content longtext
);

create table group_mission_statement(
id int auto_increment primary key,
content longtext
);

create table research_topics(
id int auto_increment primary key,
category varchar(100),
title varchar(100),
content longtext
);

create table instruments(
id int auto_increment primary key,
title varchar(100),
image longtext
)
